```python
from fastapi import FastAPI
from uuid import uuid4

app = FastAPI()
orders: dict[str, dict] = {}


@app.post("/orders")
def create_order(payload: dict):
    order_id = uuid4().hex
    orders[order_id] = {"item": payload.get("item"), "qty": payload.get("qty")}
    return {"order_id": order_id}
```
